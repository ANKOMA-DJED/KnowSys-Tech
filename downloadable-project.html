import { QueryClient } from "@tanstack/react-query";

async function throwIfResNotOk(res: Response) {
  if (!res.ok) {
    // Try to get the error message from the response
    let errMsg;
    try {
      const json = await res.json();
      errMsg = json.message || `HTTP error! status: ${res.status}`;
    } catch (e) {
      errMsg = `HTTP error! status: ${res.status}`;
    }
    throw new Error(errMsg);
  }
}

export async function apiRequest(
  method: string,
  url: string,
  body?: any,
): Promise {
  const headers: HeadersInit = {
    "Content-Type": "application/json",
  };

  const options: RequestInit = {
    method,
    headers,
    credentials: "include",
  };

  if (body) {
    options.body = JSON.stringify(body);
  }

  const res = await fetch(url, options);
  await throwIfResNotOk(res);
  const contentType = res.headers.get("content-type");
  if (contentType?.includes("application/json")) {
    return res.json();
  }
  return undefined;
}

type UnauthorizedBehavior = "returnNull" | "throw";
export const getQueryFn: (options: {
  on401: UnauthorizedBehavior;
}) => (context: { queryKey: string[] }) => Promise = (options) => async (
  context,
) => {
  const { queryKey } = context;
  const url = queryKey[0];
  const res = await fetch(url, {
    credentials: "include",
  });
  if (res.status === 401) {
    if (options.on401 === "throw") throw new Error("Unauthorized");
    if (options.on401 === "returnNull") return null;
  }
  await throwIfResNotOk(res);

  if (res.headers.get("Content-Length") === "0") {
    return null;
  }

  const contentType = res.headers.get("content-type");
  if (contentType?.includes("application/json")) {
    return res.json();
  }
  return null;
};

const RecipeCard = ({ recipe, onViewRecipe, favoriteId }: RecipeCardProps) => {
  const { toast } = useToast();
  const [isAnimating, setIsAnimating] = useState(false);

  // Check if recipe is in favorites if not already known
  const { data: favoriteStatus } = useQuery({
    queryKey: ['/api/favorites/check', recipe.id.toString()],
    queryFn: () => fetch(`/api/favorites/check/${recipe.id}`).then(res => res.json()),
    enabled: !favoriteId,
  });

  const isFavorite = favoriteId || (favoriteStatus?.isFavorite);
  const currentFavoriteId = favoriteId || (favoriteStatus?.favorite?.id);

  // Add to favorites mutation
  const addToFavorites = useMutation({
    mutationFn: async () => {
      return apiRequest('POST', '/api/favorites', {
        recipeId: recipe.id.toString(),
        recipeName: recipe.title,
        recipeImage: recipe.image,
        cookTime: recipe.readyInMinutes,
        difficulty: recipe.difficulty,
        servings: recipe.servings,
        summary: recipe.summary
      });
    },
    onSuccess: () => {
      toast({
        title: "Added to favorites",
        description: `${recipe.title} has been added to your favorites.`,
      });
      queryClient.invalidateQueries({ queryKey: ['/api/favorites'] });
      queryClient.invalidateQueries({ queryKey: ['/api/favorites/check', recipe.id.toString()] });
    },
    onError: () => {
      toast({
        title: "Error",
        description: "Failed to add recipe to favorites.",
        variant: "destructive",
      });
    },
  });

  // Remove from favorites mutation
  const removeFromFavorites = useMutation({
    mutationFn: async () => {
      return apiRequest('DELETE', `/api/favorites/${currentFavoriteId}`);
    },
    onSuccess: () => {
      toast({
        title: "Removed from favorites",
        description: `${recipe.title} has been removed from your favorites.`,
      });
      queryClient.invalidateQueries({ queryKey: ['/api/favorites'] });
      queryClient.invalidateQueries({ queryKey: ['/api/favorites/check', recipe.id.toString()] });
    },
    onError: () => {
      toast({
        title: "Error",
        description: "Failed to remove recipe from favorites.",
        variant: "destructive",
      });
    },
  });

  const toggleFavorite = () => {
    setIsAnimating(true);
    setTimeout(() => setIsAnimating(false), 1000);
    
    if (isFavorite) {
      removeFromFavorites.mutate();
    } else {
      addToFavorites.mutate();
    }
  };

    import express, { type Request, Response, NextFunction } from "express";
import { registerRoutes } from "./routes";
import { setupVite, serveStatic, log } from "./vite";

const app = express();
app.use(express.json());
app.use(express.urlencoded({ extended: false }));

app.use((req, res, next) => {
  const start = Date.now();
  const path = req.path;
  let capturedJsonResponse: Record | undefined = undefined;

  const originalResJson = res.json;
  res.json = function (bodyJson, ...args) {
    capturedJsonResponse = bodyJson;
    return originalResJson.apply(res, [bodyJson, ...args]);
  };

  res.on("finish", () => {
    const duration = Date.now() - start;
    if (path.startsWith("/api")) {
      let logLine = `${req.method} ${path} ${res.statusCode} in ${duration}ms`;
      if (capturedJsonResponse) {
        logLine += ` :: ${JSON.stringify(capturedJsonResponse)}`;
      }

      if (logLine.length > 80) {
        logLine = logLine.slice(0, 79) + "…";
      }

      log(logLine);
    }
  });

  next();
});

(async () => {
  const server = await registerRoutes(app);

  app.use((err: any, _req: Request, res: Response, _next: NextFunction) => {
    const status = err.status || err.statusCode || 500;
    const message = err.message || "Internal Server Error";

    res.status(status).json({ message });
    throw err;
  });

  // importantly only setup vite in development and after
  // setting up all the other routes so the catch-all route
  // doesn't interfere with the other routes
  if (app.get("env") === "development") {
    await setupVite(app, server);
  } else {
    serveStatic(app);
  }

  // ALWAYS serve the app on port 5000
  // this serves both the API and the client.
  // It is the only port that is not firewalled.
  const port = 5000;
  server.listen({
    port,
    host: "0.0.0.0",
    reusePort: true,
  }, () => {
    log(`serving on port ${port}`);
  });
})();

        easyOnly: req.query.easyOnly === "true",
        quickMeals: req.query.quickMeals === "true",
        cuisine: req.query.cuisine as string || "All cuisines"
      };
      
      const recipes = await storage.searchRecipes(query, filters);
      res.json(recipes);
    } catch (error) {
      console.error("Recipe search error:", error);
      res.status(500).json({ message: "Failed to search recipes" });
    }
  });

  // Get a specific recipe by ID
  app.get("/api/recipes/:id", async (req, res) => {
    try {
      const { id } = req.params;
      const recipe = await storage.getRecipeById(id);
      res.json(recipe);
    } catch (error) {
      console.error("Get recipe error:", error);
      res.status(500).json({ message: "Failed to fetch recipe details" });
    }
  });

  // Get all favorites
  app.get("/api/favorites", async (req, res) => {
    try {
      // For demo purposes, using a fixed userId of 1
      const userId = 1;
      const favorites = await storage.getFavorites(userId);
      res.json(favorites);
    } catch (error) {
      console.error("Get favorites error:", error);
      res.status(500).json({ message: "Failed to fetch favorites" });
    }
  });

  // Add a recipe to favorites
  app.post("/api/favorites", async (req, res) => {
    try {
      // Define schema for favorite
      const favoriteSchema = z.object({
        userId: z.number(),
        recipeId: z.string(),
        recipeName: z.string(),
        recipeImage: z.string().optional(),
        cookTime: z.number().optional(),
        difficulty: z.string().optional(),
        servings: z.number().optional(),
        summary: z.string().optional(),
        dateAdded: z.string()
      });
      
      // Validate request body
      const favoriteData = favoriteSchema.parse({
        ...req.body,
        userId: 1, // For demo purposes, using a fixed userId of 1
        dateAdded: new Date().toISOString()
      });
      
      // Check if recipe is already in favorites
      const existingFavorite = await storage.getFavoriteByRecipeId(favoriteData.userId, favoriteData.recipeId);
      
      if (existingFavorite) {
        return res.status(409).json({ message: "Recipe already in favorites" });
      }
      
      const favorite = await storage.addFavorite(favoriteData);
      res.status(201).json(favorite);
    } catch (error) {
      console.error("Add favorite error:", error);
      
      if (error instanceof ZodError) {
        const validationError = fromZodError(error);
        return res.status(400).json({ message: validationError.message });
      }
      
      res.status(500).json({ message: "Failed to add recipe to favorites" });
    }
  });

  // Remove a recipe from favorites
  app.delete("/api/favorites/:id", async (req, res) => {
    try {
      const { id } = req.params;
      const favorite = await storage.getFavoriteById(parseInt(id));
      
      if (!favorite) {
        return res.status(404).json({ message: "Favorite not found" });
      }
      
      await storage.removeFavorite(parseInt(id));
      res.status(204).send();
    } catch (error) {
      console.error("Remove favorite error:", error);
      res.status(500).json({ message: "Failed to remove recipe from favorites" });
    }
  });

  // Check if a recipe is in favorites
  app.get("/api/favorites/check/:recipeId", async (req, res) => {
    try {
      const { recipeId } = req.params;
      const userId = 1; // For demo purposes, using a fixed userId of 1
      
      const favorite = await storage.getFavoriteByRecipeId(userId, recipeId);
      res.json({ isFavorite: !!favorite, favorite });
    } catch (error) {
      console.error("Check favorite error:", error);
      res.status(500).json({ message: "Failed to check favorite status" });
    }
  });

  const httpServer = createServer(app);

  return httpServer;

export interface IStorage {

export class MemStorage implements IStorage {
  private users: Map;
  private favoritesList: Map;
  currentUserId: number;
  currentFavoriteId: number;
  
  constructor() {
    this.users = new Map();
    this.favoritesList = new Map();
    this.currentUserId = 1;
    this.currentFavoriteId = 1;
  }

  async getUser(id: number): Promise {
    return this.users.get(id);
  }

  async getUserByUsername(username: string): Promise {
    return Array.from(this.users.values()).find(
      (user) => user.username === username,
    );
  }

  async createUser(insertUser: any): Promise {
    const id = this.currentUserId++;
    const user = { ...insertUser, id };
    this.users.set(id, user);
    return user;
  }
  
  // Favorites methods
  async getFavorites(userId: number): Promise {
    return Array.from(this.favoritesList.values()).filter(
      favorite => favorite.userId === userId
    );
  }
  
  async getFavoriteById(id: number): Promise {
    return this.favoritesList.get(id);
  }
  
  async getFavoriteByRecipeId(userId: number, recipeId: string): Promise {
    return Array.from(this.favoritesList.values()).find(
      favorite => favorite.userId === userId && favorite.recipeId === recipeId
    );
  }
  
  async addFavorite(insertFavorite: InsertFavorite): Promise {
    const id = this.currentFavoriteId++;
    const favorite: Favorite = { ...insertFavorite, id };
    this.favoritesList.set(id, favorite);
    return favorite;
  }
  
  async removeFavorite(id: number): Promise {
    this.favoritesList.delete(id);
  }
  
  // Recipe API methods
  async searchRecipes(query: string, filters: any): Promise {
    try {
      const apiKey = process.env.SPOONACULAR_API_KEY || '';
      if (!apiKey) {
        throw new Error('API key not found');
      }
      
      const params: Record = {
        apiKey,
        query,
        number: 12,
      };
      
      // Add filter parameters
      if (filters.easyOnly) {
        params.maxReadyTime = 45; // Easy recipes are typically quick to make
      }
      
      if (filters.quickMeals) {
        params.maxReadyTime = 30;
      }
      
      if (filters.cuisine && filters.cuisine !== 'All cuisines') {
        params.cuisine = filters.cuisine;
      }
      
      const response = await axios.get('https://api.spoonacular.com/recipes/complexSearch', {
        params: {
          ...params,
          addRecipeInformation: true,
          fillIngredients: true,
        }
      });
      
      // Process results to add difficulty level based on cooking time
      const results = response.data.results.map((recipe: any) => {
        let difficulty = 'medium';
        if (recipe.readyInMinutes <= 30) {
          difficulty = 'easy';
        } else if (recipe.readyInMinutes > 60) {
          difficulty = 'hard';
        }
        
        return {
          ...recipe,
          difficulty
        };
      });
      
      // If easyOnly filter is applied, filter results client-side to only include easy recipes
      const filteredResults = filters.easyOnly 
        ? results.filter((recipe: any) => recipe.difficulty === 'easy')
        : results;
        
      return {
        ...response.data,
        results: filteredResults
      };
    } catch (error) {
      console.error('Error searching recipes:', error);
      throw error;
    }
  }
  
  async getRecipeById(id: string): Promise {
    try {
      const apiKey = process.env.SPOONACULAR_API_KEY || '';
      if (!apiKey) {
        throw new Error('API key not found');
      }
      
      const response = await axios.get(`https://api.spoonacular.com/recipes/${id}/information`, {
        params: {
          apiKey,
          includeNutrition: true,
        }
      });
      
      // Add difficulty level based on cooking time
      let difficulty = 'medium';
      if (response.data.readyInMinutes <= 30) {
        difficulty = 'easy';
      } else if (response.data.readyInMinutes > 60) {
        difficulty = 'hard';
      }
      
      return {
        ...response.data,
        difficulty
      };
    } catch (error) {
      console.error('Error fetching recipe:', error);
      throw error;
    }
  }
}
